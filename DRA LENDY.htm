        const generateBtn = document.getElementById('generateBtn');

        const selectElement = document.getElementById('conditionSelect');

        const resultContainer = document.getElementById('resultContainer');

        const loadingIndicator = document.getElementById('loadingIndicator');

        const API_URL = "https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=";

 

        generateBtn.addEventListener('click', async () => {

            const selectedCondition = selectElement.value;

            const prompt = `Actúa como un médico experto. Escribe una explicación sencilla y amigable sobre "${selectedCondition}", como si se la estuvieras explicando a un padre. No uses jerga médica compleja. La explicación debe ser un resumen de 2 o 3 oraciones. Responde directamente con el texto.`;

            resultContainer.innerHTML = '';

            loadingIndicator.classList.remove('hidden');

 

            const payload = {

                contents: [{ parts: [{ text: prompt }] }]

            };

 

            const apiKey = "";

            let retries = 0;

            const maxRetries = 5;

            let response;

 

            while (retries < maxRetries) {

                try {

                    response = await fetch(`${API_URL}${apiKey}`, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });

                    if (response.ok) {

                        break;

                    } else if (response.status === 429) {

                        retries++;

                        await new Promise(res => setTimeout(res, Math.pow(2, retries) * 1000));

                    } else {

                        throw new Error(`Error: ${response.status}`);

                    }

                } catch (error) {

                    console.error('Fetch error:', error);

                    break;

                }

            }

 

            loadingIndicator.classList.add('hidden');

            if (!response || !response.ok) {

                resultContainer.innerHTML = '<p class="text-red-500">Lo sentimos, no pudimos generar la información en este momento. Inténtalo de nuevo más tarde.</p>';

                return;

            }

 

            const result = await response.json();

            if (result.candidates && result.candidates.length > 0 && result.candidates[0].content && result.candidates[0].content.parts && result.candidates[0].content.parts.length > 0) {

                const text = result.candidates[0].content.parts[0].text;

                resultContainer.innerHTML = `<p>${text}</p>`;

            } else {

                resultContainer.innerHTML = '<p class="text-red-500">No se pudo obtener una respuesta válida. Inténtalo de nuevo.</p>';

            }

        });

 

        function wrapLabels(label, maxWidth) {

            const words = label.split(' ');

            let lines = [];

            let currentLine = '';

            words.forEach(word => {

                if ((currentLine + word).length > maxWidth) {

                    lines.push(currentLine.trim());

                    currentLine = '';

                }

                currentLine += word + ' ';

            });

            lines.push(currentLine.trim());

            return lines;

        }

 

        const tooltipTitleCallback = (tooltipItems) => {

            const item = tooltipItems[0];

            let label = item.chart.data.labels[item.dataIndex];

            if (Array.isArray(label)) {

                return label.join(' ');

            }

            return label;

        };

       

        const supportCtx = document.getElementById('supportNetworkChart').getContext('2d');
